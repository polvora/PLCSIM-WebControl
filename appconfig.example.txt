# PLCSIM-AutoStart configuration template.
# The installer copies this to appconfig.txt next to the executable.
# The service rewrites appconfig.txt when you change settings from the web UI,
# so edit it manually only while the service is STOPPED.
# A blank value means "use the built-in default".
# Full reference: docs/CONFIGURATION.md

# ---- Web server ----
# URL the service listens on. Default http://+:8090/ = all interfaces (reachable from the LAN),
# because remote control is the main feature. The installer also opens the firewall for the port.
# Use http://localhost:8090/ instead if you want local-only access.
http_prefix = http://+:8090/

# ---- Workspace ----
# Folder that contains your PLCSIM workspaces. Blank = <Documents>\PLCSIM of the account running the service.
workspace_root =
# Active workspace folder. Blank = auto-detect the most recently used one.
workspace =

# ---- Instance auto-start ----
autostart_enabled = false
# last  = restore whatever was powered on before the last shutdown.
# fixed = always power on the single instance named in autostart_instance.
autostart_mode = last
autostart_instance =

# ---- Power-on limit ----
# The most PLCs you can power on by hand at once (also editable in the UI).
max_powered_on = 1
# The auto-start cap: the most PLCs auto-start turns on at boot (also editable in the UI).
# Set it to how many this machine can run at once. Only affects auto-start; the manual limit can be higher.
hard_max_powered_on = 1

# ---- Boot protections (loop-breaker) ----
# How many boots in a row can fail to come up working before the service stops auto-starting (SAFE MODE).
boot_fail_limit = 2
# How long (seconds) after auto-start the machine must keep responding before the boot counts as good.
stable_seconds = 90
# Pause (ms) between power-ons during auto-start, so they don't all start at the same time.
start_stagger_ms = 4000
# How often (ms) the tool checks its own web page, and how long a check may take before it counts as
# failed (the sign of a machine that looks alive but is frozen).
health_probe_interval_ms = 15000
health_timeout_ms = 3000
# Internal state; leave blank.
last_running =

# ---- Network ----
# Softbus = zero-config simulation bus (default). TCPIPSingleAdapter / TCPIPMultipleAdapter = real Ethernet.
network_mode = Softbus
# Host adapter ifIndex + name (only used for TCP/IP modes; pick it from the web UI).
adapter_index = 0
adapter_name =

# ---- Advanced ----
# Instance storage layout. "default" uses PLCSIM's native persistent storage (recommended).
storage_layout = default
poweron_timeout = 60000
run_timeout = 60000
# How long to wait for the PLCSIM runtime manager at startup (seconds).
connect_wait_seconds = 120
# Explicit path to Siemens.Simatic.Simulation.Runtime.Api.x64.dll. Blank = auto-detect the installed PLCSIM Advanced.
api_dll_path =
# Log file path. Blank = autostart.log next to the executable.
log =
